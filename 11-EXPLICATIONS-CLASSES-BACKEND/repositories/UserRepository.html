UserRepository.cs - Repository Utilisateurs

COUCHE 2 : REPOSITORIES • Gestion CRUD des utilisateurs

Role — UserRepository encapsule toutes les operations d'acces aux donnees pour l'entite User. C'est la seule classe qui communique directement avec la table Users via Entity Framework.

Code Source Complet

using Microsoft.EntityFrameworkCore;
using XtraWork.Entities;

namespace XtraWork.Repositories;

public class UserRepository
{
    private readonly XtraWorkContext _context;

    public UserRepository(XtraWorkContext context)
    {
        _context = context;
    }

    public async Task<User?> GetByIdAsync(Guid id)
    {
        return await _context.Users.FindAsync(id);
    }

    public async Task<User?> GetByUsernameAsync(string username)
    {
        return await _context.Users
            .FirstOrDefaultAsync(u => u.Username == username);
    }

    public async Task<User?> GetByEmailAsync(string email)
    {
        return await _context.Users
            .FirstOrDefaultAsync(u => u.Email == email);
    }

    public async Task<User> CreateAsync(User user)
    {
        _context.Users.Add(user);
        await _context.SaveChangesAsync();
        return user;
    }

    public async Task<User> UpdateAsync(User user)
    {
        _context.Users.Update(user);
        await _context.SaveChangesAsync();
        return user;
    }

    public async Task<List<User>> GetAllAsync()
    {
        return await _context.Users.ToListAsync();
    }

    public async Task<List<User>> GetByRoleAsync(string role)
    {
        return await _context.Users
            .Where(u => u.Role == role)
            .ToListAsync();
    }
}

Methodes Principales

GetByUsernameAsync() - Recherche par username

Utilisation — Utilisee dans AuthService.LoginAsync() pour chercher l'utilisateur lors de la connexion.

• FirstOrDefaultAsync() : retourne le premier ou null
• u => u.Username == username : expression lambda (filtre LINQ)
• SQL genere : SELECT TOP(1) * FROM Users WHERE Username = @p0

GetByEmailAsync() - Recherche par email

Verification unicite — Utilisee dans RegisterAsync() pour verifier qu'un email n'est pas deja utilise avant de creer un nouvel utilisateur.

CreateAsync() - Creer un utilisateur

Insertion en BDD
• Add(user) : marque l'entite pour insertion
• SaveChangesAsync() : execute l'INSERT SQL
• SQL : INSERT INTO Users (...) VALUES (...)

GetByRoleAsync() - Filtrer par role

Filtrage — Permet de recuperer tous les utilisateurs d'un role specifique.

Exemples : GetByRoleAsync("Admin"), GetByRoleAsync("Manager")

Exemples Pratiques

Verifier si username existe

var existingUser = await userRepository.GetByUsernameAsync("marie.user");
if (existingUser != null)
{
    throw new InvalidOperationException("Ce nom d'utilisateur existe deja");
}

Lister tous les admins

var admins = await userRepository.GetByRoleAsync("Admin");
Console.WriteLine($"Il y a {admins.Count} administrateurs");
