public class AuthService
{
    private readonly UserRepository _userRepository;
    private readonly IConfiguration _configuration;

    public async Task<AuthResponse> LoginAsync(LoginRequest request)
    {
        var user = await _userRepository.GetByUsernameAsync(request.Username);
        
        if (user == null || !BCrypt.Net.BCrypt.Verify(request.Password, user.PasswordHash))
        {
            throw new UnauthorizedAccessException("Nom d'utilisateur ou mot de passe incorrect");
        }

        if (!user.IsActive)
        {
            throw new UnauthorizedAccessException("Compte desactive");
        }

        user.LastLoginAt = DateTime.UtcNow;
        await _userRepository.UpdateAsync(user);

        var token = GenerateJwtToken(user);
        return new AuthResponse { Token = token, User = ... };
    }

    private string GenerateJwtToken(User user)
    {
        var claims = new[]
        {
            new Claim(ClaimTypes.NameIdentifier, user.Id.ToString()),
            new Claim(ClaimTypes.Name, user.Username),
            new Claim(ClaimTypes.Email, user.Email),
            new Claim(ClaimTypes.Role, user.Role)
        };
        
        var token = new JwtSecurityToken(claims: claims, expires: DateTime.UtcNow.AddHours(24));
        return new JwtSecurityTokenHandler().WriteToken(token);
    }
}

Methodes Principales

LoginAsync() — Processus : 1) Chercher user par username, 2) Verifier password avec BCrypt.Verify(), 3) Verifier IsActive, 4) Mettre a jour LastLoginAt, 5) Generer token JWT, 6) Retourner token + infos user.
RegisterAsync() — Processus : 1) Verifier username unique, 2) Verifier email unique, 3) Hasher le password avec BCrypt, 4) Creer User avec Role = "User", 5) Generer token, 6) Retourner.
GenerateJwtToken() — Cree un token JWT contenant les claims (NameIdentifier, Name, Email, Role). Expire apres 24h. Signe avec la cle secrete de appsettings.json.
